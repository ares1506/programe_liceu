      int div, p1=0, p2=0, x=2, k=0;

      if((n==1)||(m==1))

          return 0;

      while((n>1)&&(m>1)){

          div=0;

          for(int i=1;i<=x;i++)

              if(x%i==0)

                  div++;

          if(div==2){

              while(n%x==0)

                  n/=x,p1++;

              while(m%x==0)

                  m/=x,p2++;

              if((p1==p2)&&(p1!=0)&&(p2!=0))

                  k++;

          }

          p1=0,p2=0,x++;

      }

      return k;

  }
